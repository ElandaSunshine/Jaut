49 bool xmlTestAndSkipSequence(juce::String::CharPointerType &it)
50 {
51 auto c_it = it;
52
53 for (;;)
54 {
55 const juce::juce_wchar c = *c_it;
56
57 if (c == 0 || c == ' ')
58 {
59 return false;
60 }
61 else if (c == ';')
62 {
63 it = (++c_it);
64 return true;
65 }
66
67 ++c_it;

71 juce::String encodeString(const juce::String &string)
72 {
73 juce::String::CharPointerType it = string.getCharPointer();
74 juce::String output;
75 output.preallocateBytes(string.getNumBytesAsUTF8() * JUCE_STRING_UTF_TYPE);
76
77 for (;;)
78 {
79 const juce::juce_wchar c = *it;
80
81 if (c == 0)
82 {
83 break;
84 }
85
86 switch (c)
87 {
88 case '\"': output << "&quot;"; break;
89 case '\'': output << "&apos;"; break;
90 case '<': output << "&lt;"; break;
91 case '>': output << "&gt;"; break;
92
93 case '&':
94 {
95 if (!xmlTestAndSkipSequence(it))
96 {
97 output << "&amp;";
98 break;
99 }
100 }
101
102 default: output << c;
103 }
104
105 ++it;
106 }
107
108 return output.replace("\n", "\\n");
109 }

235 juce::var xmlNodeToVar(const juce::XmlElement &element, const jaut::XmlParser::Options &options)
236 {
237 const juce::String &tag_name = element.getTagName();
238
239 if (element.getBoolAttribute("xsi:nil") || tag_name == options.typeTags.tagTypeArrayVoid)
240 {
241 return {};
242 }
243
244 if (tag_name == options.typeTags.tagTypeString) return ::decodeString(element.getAllSubText());
245
246 const juce::String content = element.getAllSubText().trim();
247
248 if (tag_name == options.typeTags.tagTypeInt) return content.getIntValue();
249 if (tag_name == options.typeTags.tagTypeInt64) return content.getLargeIntValue();
250 if (tag_name == options.typeTags.tagTypeDouble) return content.getDoubleValue();
251 if (tag_name == options.typeTags.tagTypeBool) return static_cast<bool>(juce::var(content));
252 if (tag_name == options.typeTags.tagTypeJson) return juce::JSON::parse(content);
253
254 if (tag_name == options.typeTags.tagTypeArray)
255 {
256 juce::Array<juce::var> array;
257
258 for (const auto &array_element : element.getChildIterator())
259 {
260 juce::var value = ::xmlNodeToVar(*array_element, options);
261
262 if (!value.isUndefined())
263 {
264 array.add(std::move(value));
265 }
266 }
267
268 return array;
269 }
270
271 if (options.throwOnInvalidTagName)
272 {
273 throw jaut::ConfigParseException("Invalid type-tag '" + tag_name + "'");
274 }
275
276 return juce::var::undefined();
277 }

282 ContentGenerator generator,
283 jaut::detail::Indent indent,

300 output << ' ';

312 if (text.containsChar('\n'))

315 }

317 {
318 output << text;
319 }
320
321 output << "</" << name << '>';
322 }
323 else

325 output << "/>";
326 }

331 juce::String xmlGenerateComment(const juce::String &comment, const juce::String &indent)
332 {
333 const juce::String trimmed = comment.trim();
334 juce::String output;
335
336 if (!trimmed.isEmpty())
337 {
338 const juce::String comment_escaped = trimmed.replace("-->", "- ->");
339
340 if (comment_escaped.containsChar('\n'))
341 {
342 const juce::StringArray lines = juce::StringArray::fromLines(comment_escaped);
343 const juce::String space = indent + " ";
344 output << indent << "<!-- " << lines.getReference(0) << '\n';
345
346 for (auto it = (lines.begin() + 1); it != lines.end(); ++it)
347 {
348 output << space << *it;
349
350 if (std::next(it) != lines.end())
351 {
352 output << '\n';
353 }
354 }
355
356 output << " -->";
357 }
358 else
359 {
360 output << indent + "<!-- " + comment_escaped + " -->";
361 }
362 }
363
364 return output;

489 jaut::Config::Property &propertyObj,

493 {

504
505 if (tag_name == options.typeTags.tagTypeTree)
506 {
507 if (prop->isValueProperty())

509 continue;

518 continue;

522

545 void XmlParser::parseConfig(ParseAttributes<Config::CategoryMap> parAttributes)
546 {
547 juce::XmlDocument document(parAttributes.configFile);
548 const std::unique_ptr<juce::XmlElement> config = document.getDocumentElement();
549
550 if (!config)
551 {
552 throw ConfigParseException(document.getLastParseError());
553 }
554
555 if (!config->hasTagName(options.typeTags.tagNameRoot))
556 {
557 throw ConfigParseException("invalid root tag-name '" + config->getTagName() + "' where there "
558 "was '" + options.typeTags.tagNameRoot + "' expected");
559 }
560
561 jaut::Config::CategoryMap &categories = parAttributes.categories;
562
563 for (const auto &xml_entity : config->getChildIterator())
564 {
565 if (!xml_entity->hasTagName(options.typeTags.tagNameCategory))
566 {
567 if (options.throwOnInvalidTagName)
568 {
569 throw jaut::ConfigParseException("invalid tag-name '" + xml_entity->getTagName() + "' where there "
570 "was '" + options.typeTags.tagNameCategory + "' expected");
571 }
572
573 continue;
574 }
575
576 if (!xml_entity->hasAttribute("name"))
577 {
578 continue;
579 }
580
581 const juce::String &cat_id = xml_entity->getStringAttribute("name");
582
583 if (const auto cat_prop_it = categories.find(cat_id); cat_prop_it != categories.end())
584 {
585 ::xmlReadProperty(*xml_entity, cat_prop_it->second, options);
586 }
587 }

591 {
592 const detail::Indent indent{ 0, options.indentationWidth };
593 const juce::String indent_p0 = indent.toString();
594 const juce::String notice = ::xmlGenerateComment(parAttributes.options.configNotice, indent_p0);

603 {

607 detail::transformCategories(parAttributes.categories, categories, cat_default);
608
609 const jaut::Config::Property &prop_cat_default = parAttributes.categories.at(cat_default);
610 lOutput << ::xmlGenerateCategory(prop_cat_default, options) << '\n';
611
612 for (auto it = categories.begin(); it != categories.end(); ++it)
613 {
614 if (it != categories.begin())
615 {
616 const bool has_comment = !it->get().second.getComment().trim().isEmpty();
617 auto l_it = std::prev(it);
618
619 if (has_comment || !l_it->get().second.getComment().trim().isEmpty())
620 {
621 lOutput << '\n';
622 }
623 }
624 else if (!prop_cat_default.getComment().trim().isEmpty())
625 {
626 lOutput << '\n';
627 }
628
629 lOutput << ::xmlGenerateCategory(it->get().second, options);
630
631 if (std::next(it) != categories.end())
632 {
633 lOutput << '\n';
634 }
635 }
636 };
637
638 const auto att_xsi = std::make_pair("xmlns:xsi", "https://www.w3.org/2001/XMLSchema-instance");
639 output << ::xmlGenerateTag(options.typeTags.tagNameRoot, category_generator, indent, { att_xsi }) << '\n';
640
641 if (!parAttributes.configFile.replaceWithText(output))
