123 bool ContentPane::updateComponent(ContentPtr parContentPtr)
124 {
125 if (parContentPtr == content)
126 {
127 return false;
128 }
129
130 std::swap(content, parContentPtr);
131 removeAllChildren();
132
133 if (content)
134 {
135 addAndMakeVisible(content.get());
136 resizeContents();
137 }
138
139 eventContentChanged.invoke(content.get());
140 return true;
141 }
