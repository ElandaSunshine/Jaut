297 juce::String jsonStripCommentsAndTrailingCommas(const juce::String &jsonIn)
298 {
299 juce::String out;
300 out.preallocateBytes(jsonIn.getCharPointer().sizeInBytes());
301
302 juce::String::CharPointerType it = jsonIn.getCharPointer();
303
304 for (;;)
305 {
306 const juce::juce_wchar c = it.getAndAdvance();
307
308 if (c == '\0')
309 {
310 break;
311 }
312
313 if (c == '"')
314 {
315 out << c;
316 stripCommentsParseQuotedString(out, it);
317 }
318 else if (c == '/')
319 {
320 const juce::juce_wchar c_2 = it.getAndAdvance();
321
322 if (c_2 == '\0')
323 {
324 break;
325 }
326
327 if (c_2 == '/') // process single line
328 {
329 stripCommentsSkipUntil(it, '\n');
330 }
331 else if (c_2 == '*') // process multi-line
332 {
333 stripCommentsParseMultiline(it);
334 }
335 else
336 {
337 out << c_2;
338 }
339 }
340 else if (c == ',')
341 {
342 if (stripCommentsCommaFollowedByObjectEnd(it))
343 {
344 continue;
345 }
346
347 out << c;
348 }
349 else
350 {
351 out << c;
352 }
353 }
354
355 return out;
