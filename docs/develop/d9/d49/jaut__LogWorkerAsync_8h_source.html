204 inline ILogWorker::FlushAttemptResult LogWorkerAsync<N, L>::tryFlush(const LogMessage &parLastMessage)
205 {
206 if (!logger)
207 {
208 return ILogWorker::FlushAttemptResult::NotReady;
209 }
210
211 if (dirty)
212 {
213 return ILogWorker::FlushAttemptResult::Async;
214 }
215
216 if ( flushBehaviour.policies.test(FlushPolicy::Instant)
217 || (flushBehaviour.policies.test(FlushPolicy::Filled) && isFull())
218 || (flushBehaviour.policies.test(FlushPolicy::Levelled) && parLastMessage.level >= flushBehaviour.level)
219 || (flushBehaviour.policies.test(FlushPolicy::Custom) && flushBehaviour.customPolicy(parLastMessage)))
220 {
221 flush();
222
223 if (!dirty)
224 {
225 return ILogWorker::FlushAttemptResult::Successful;
226 }
227 }
228
229 return ILogWorker::FlushAttemptResult::Async;
230 }

233 template<int N, class L>
234 inline void LogWorkerAsync<N, L>::run()
235 {
236 while (running)
237 {
238 if (flushBehaviour.policies.test(FlushPolicy::Timed))
239 {
240 const TimePoint now = Clock::now();
241 const int elapsed = static_cast<int>(std::chrono::duration_cast<std::chrono::seconds>(now - lastTime)
242 .count());
243
244 if (elapsed >= flushBehaviour.interval)
245 {
246 dirty = true;
247 lastTime = Clock::now();
248 }
249 }
250
251 if (dirty.exchange(false))
252 {
253 if (buffer.isEmpty())
254 {
255 continue;
256 }
257
258 std::vector<LogMessage> messages;
259 messages.reserve(buffer.size() + 10);
260
261 while (!buffer.isEmpty())
262 {
263 messages.emplace_back(buffer.pop());
264 }
265
266 std::stable_sort(messages.begin(), messages.end(),
267 [](auto &&left, auto &&right) { return (left.timestamp < right.timestamp);});
268
269 for (const AbstractLogger::SinkPtr &sink_ptr : logger->getSinks())
270 {
271 sink_ptr->prepare(static_cast<int>(messages.size()));
272
273 for (LogMessage &message : messages)
274 {
275 sink_ptr->print(message);
276 }
277
278 sink_ptr->flush();
279 }
280 }
281
282 #if JAUT_PROVIDER_LOGGER_ASYNC_SLEEP > -1
283 // Cut some other threads some slack,
284 // the logger thread is not that important that it has to consume every cycle it gets
285 // we don't really care if the delay is a tad bit more than specified
286 sleep(JAUT_LOGGER_ASYNC_SLEEP);
287 #endif
288 }
289
290 for (const AbstractLogger::SinkPtr &sink_ptr : logger->getSinks())
291 {
292 sink_ptr->onClose();
293 }
294 }
