187 template<class First, class ...Remainder>
188 static juce::var getOpt(const Node &node)
189 {
190 const auto opt = node.as<std::optional<First>>(std::nullopt);
191
192 if (opt.has_value())
193 {
194 return opt.value();
195 }
196
197 if constexpr (sizeof...(Remainder) > 0)
198 {
199 return getOpt<Remainder...>(node);
200 }
201 else
202 {
203 return juce::var::undefined();
204 }
205 }

207 //==============================================================================================================
208 static Node encode(const juce::var& rhs)
209 {
210 switch (jaut::VarUtil::getVarType(rhs))
211 {
212 case jaut::VarUtil::VarTypeId::String: return Node(rhs.toString());
213 case jaut::VarUtil::VarTypeId::Bool: return Node(static_cast<bool>(rhs));
214 case jaut::VarUtil::VarTypeId::Double: return Node(static_cast<double>(rhs));
215 case jaut::VarUtil::VarTypeId::Int: return Node(static_cast<int>(rhs));
216 case jaut::VarUtil::VarTypeId::Int64: return Node(static_cast<juce::int64>(rhs));
217 case jaut::VarUtil::VarTypeId::Void: return Node(NodeType::Null);
218 case jaut::VarUtil::VarTypeId::Array: return Node(*rhs.getArray());
219 case jaut::VarUtil::VarTypeId::DynamicObject: return Node(rhs.getDynamicObject()->getProperties());
220
221 case jaut::VarUtil::VarTypeId::Undefined:
222 case jaut::VarUtil::VarTypeId::OtherObject:
223 case jaut::VarUtil::VarTypeId::Method:
224 case jaut::VarUtil::VarTypeId::BinaryData:
225 case jaut::VarUtil::VarTypeId::Unknown:
226 break;
227 }
228
229 return Node(NodeType::Undefined);
230 }

232 static bool decode(const Node& node, juce::var& rhs)
233 {
234 switch (node.Type())
235 {
236 case NodeType::Undefined: rhs = juce::var::undefined(); break;
237 case NodeType::Null: rhs = juce::var(); break;
238 case NodeType::Sequence: rhs = node.as<juce::Array<juce::var>>(); break;
239
240 case NodeType::Scalar:
241 {
242 rhs = getOpt<int, juce::int64, double, bool, juce::String>(node);
243 break;
244 }
245
246 case NodeType::Map:
247 {
248 auto d_obj = std::make_unique<juce::DynamicObject>();
249 d_obj->getProperties() = node.as<juce::NamedValueSet>();
250 rhs = d_obj.release();
251
252 break;
253 }
254 }
255
256 return true;
257 }
