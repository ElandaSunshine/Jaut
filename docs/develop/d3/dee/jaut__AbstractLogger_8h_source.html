404 inline AbstractLogger::LogBuilder& AbstractLogger::LogBuilder::operator<<(T &&parObject)
405 {
406 using Type = std::decay_t<T>;
407
408 if (logMessage.level != Level::Off)
409 {
410 if constexpr (std::is_same_v<Type, LogMessage::Field>)
411 {
412 detail::emplaceField(logMessage.fields, std::forward<T>(parObject));
413 }
414 else if constexpr (std::is_base_of_v<std::exception, Type>)
415 {
416 logMessage.exception = LogMessage::ExceptionSpec::fromException(std::forward<T>(parObject));
417 }
418 else
419 {
420 logMessage.message << jaut::toString(std::forward<T>(parObject));
421 }
422 }
423
424 return *this;
425 }

477 public:
478 struct Result
479 {
480 FieldVec fields;
481 std::optional<LogMessage::ExceptionSpec> exceptionSpec { std::nullopt };
482 };

484 //==========================================================================================================
485 static void processArg(Result &result, LogMessage::Field field)
486 {
487 detail::emplaceField(result.fields, std::move(field));
488 }
489
490 static void processArg(Result &result, const std::exception &ex)
491 {
492 result.exceptionSpec = LogMessage::ExceptionSpec::fromException(ex);
493 }

495 //==========================================================================================================
496 template<class ...Fields>
497 Result operator()(Fields &&...fields)
498 {
499 Result result;
500 result.fields.reserve(sizeof...(Fields));
501
502 (processArg(result, std::forward<Fields>(fields)), ...);
503 result.fields.shrink_to_fit();

506 }

530 {
531 using Filter = ArgFilter<
532 PredicateOr<
533 std::is_same<std::decay_t<PType<>>, LogMessage::Field>,
534 std::is_base_of<std::exception, std::decay_t<PType<>>>
535 >,
536 Args...
537 >;

540
541 Result result = Filter::invokeFiltered(detail::FilterArgProcessor{}, std::forward<Args>(args)...);
542 juce::String text = Filter::invokeExcluded(detail::FormatArgProcessor{message}, std::forward<Args>(args)...);
543
544 log(makeLog(level, std::move(text), std::move(result.fields), std::move(result.exceptionSpec)));
