210 void ExtraComponent::componentMovedOrResized(juce::Component &component, bool wasMoved, bool wasResized)
211 {
212 if (&component != this || internalResizeCall)
213 {
214 return;
215 }
216
217 const juce::Rectangle<int> pre_bounds = component.getBounds();
218
219 if (wasResized)
220 {
221 actualBounds.setSize(pre_bounds.getWidth(), pre_bounds.getHeight());
222 }
223
224 if (wasMoved)
225 {
226 actualBounds.setPosition(pre_bounds.getPosition());
227 }
228
229 {
230 jdscoped juce::ScopedValueSetter(internalResizeCall, true);
231 const juce::Rectangle new_bounds = style.margin.trimRectangle(actualBounds);
232
233 if (new_bounds != component.getBounds())
234 {
235 component.setBounds(new_bounds);
236 }
237 else
238 {
239 resized();
240 }
241 }
242 }
