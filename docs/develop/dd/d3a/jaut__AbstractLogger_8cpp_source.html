180 jaut::LogMessage AbstractLogger::makeLog(jaut::LogLevel::Value level,
181 juce::String message,
182 std::vector<jaut::LogMessage::Field> fields,
183 std::optional<LogMessage::ExceptionSpec> ex)
184 {
185 jaut::LogMessage log_message;
186 log_message.name = getName();
187 log_message.timestamp = juce::Time::getCurrentTime();
188 log_message.level = level;
189 log_message.threadId = std::this_thread::get_id();
190
191 if (const juce::Thread *c_thread = juce::Thread::getCurrentThread())
192 {
193 log_message.threadName = c_thread->getThreadName();
194 }
195
196 std::swap(log_message.fields, fields);
197 std::swap(log_message.message, message);
198
199 if (ex.has_value())
200 {
201 std::swap(log_message.exception, ex);
202 }
203
204 return log_message;
205 }
