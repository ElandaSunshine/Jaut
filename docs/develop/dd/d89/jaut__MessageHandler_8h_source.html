394 template<int N, int M>
395 inline void MessageHandler<N, M>::sendMessage(MessagePointer parMessage)
396 {
397 // This check will lock, so you wouldn't call this method on a real-time thread
398 // Use deferred messages instead
399 if (juce::MessageManager::getInstance()->isThisTheMessageThread())
400 {
401 if (messageBuffer.push(std::move(parMessage)) < 0)
402 {
403 throw QueueSpaceExceededException("Could not enqueue message for the target thread, queue was full");
404 }
405 }
406 else
407 {
408 if (backBuffer.push({ true, std::move(parMessage) }) < 0)
409 {
410 throw QueueSpaceExceededException("Could not enqueue message for the message thread, queue was full");
411 }
412 }
413 }

416 template<int N, int M>
417 inline void MessageHandler<N, M>::processMessages(int parCount)
418 {
419 // Do not handle messages on the message thread
420 jassert(!juce::MessageManager::getInstance()->isThisTheMessageThread());
421
422 if (cancelUpdates.load())
423 {
424 deleteUnusedMessages();
425 return;
426 }
427
428 while ((parCount--) > 0)
429 {
430 MessagePointer message = messageBuffer.pop();
431
432 if (!message)
433 {
434 break;
435 }
436
437 const bool is_deferred = (message->getDeferId() >= 0);
438 bool defer = false;
439
440 if (!is_deferred)
441 {
442 handleMessage(message.get(), MessageDirection::TargetThread);
443 }
444 else
445 {
446 defer = deferredMessageInitHandler(message.get());
447 }
448
449 if (options.enableGarbageCollecting || defer)
450 {
451 backBuffer.push({ defer, std::move(message) });
452 }
453 }
