555 inline juce::Path DistanceUtil::borderPathFromThickness(const juce::Rectangle<T> &area, Thickness<T> thickness)
556 {
557 juce::Path path;
558
559 path.addRectangle(area.withWidth (thickness.left));
560 path.addRectangle(area.withHeight(thickness.top));
561 path.addRectangle(area.withWidth(thickness.right) .withX(area.getX() + area.getWidth() - thickness.right));
562 path.addRectangle(area.withHeight(thickness.bottom).withY(area.getY() + area.getHeight() - thickness.bottom));
563
564 return path;
