152 template<class T>
153 struct JAUT_API Stringable
154 {
155 JAUT_NODISCARD
156 static juce::String toString(const T &object)
157 {
158 using Type = std::decay_t<T>;
159
160 if constexpr (std::is_constructible_v<juce::var, T>)
161 {
162 return jaut::toString(juce::var(object));
163 }
164 else if constexpr (std::is_constructible_v<juce::String, T>)
165 {
166 return juce::String(object);
167 }
168 else if constexpr (std::is_base_of_v<std::exception, Type> || std::is_same_v<Type, std::exception>)
169 {
170 return juce::String(object.what());
171 }
172 else
173 {
174 return name() + "@" + jaut::toString(std::addressof(object));
175 }
176 }
