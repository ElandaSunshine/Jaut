230 template<class T, class U, class Fn>
231 JAUT_NODISCARD
232 inline SafeFloat::Result<T> safeFloatOperation(T left, U right, Fn &&func)
233 {
234 static_assert(std::is_floating_point_v<T> || std::is_floating_point_v<U>,
235 JAUT_ASSERT_SAFEFLOAT_NO_FLOAT_TYPE);
236
237 std::feclearexcept(FE_OVERFLOW);
238
239 const T result = (std::forward<Fn>(func))(left, right);
240
241 if (std::fetestexcept(FE_OVERFLOW))
242 {
243 return {1, result};
244 }
245
246 return {0, result};
247 }
248 }
249}

261 static_assert(std::is_floating_point_v<U>, JAUT_ASSERT_SAFEFLOAT_NO_FLOAT_TYPE);
