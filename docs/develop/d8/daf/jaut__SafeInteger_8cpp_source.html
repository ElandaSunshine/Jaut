125 template<class T, class U>
126 jaut::SafeInteger::Result<T> calculate(SafeInt<T>&(SafeInt<T>::*func)(U), T left, U right)
127 {
128 SafeInt<T> result { left };
129 int code {};
130
131 try
132 {
133 (result.*func)(right);
134 }
135 catch (const SafeIntException &ex)
136 {
137 code = static_cast<int>(ex.m_code);
138 }
139
140 return { code, static_cast<T>(result) };
141 }
142}
