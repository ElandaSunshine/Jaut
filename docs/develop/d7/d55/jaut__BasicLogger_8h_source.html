281 template<class T>
282 void BasicLogger<T>::log(LogMessage parLogMessage)
283 {
284 if (!shouldLog(parLogMessage.level))
285 {
286 return;
287 }
288
289 const bool queue_result = worker.enqueue (parLogMessage);
290 const ILogWorker::FlushAttemptResult flush_result = worker.tryFlush(parLogMessage);
291
292 if (!queue_result)
293 {
294 switch (flush_result)
295 {
296 case ILogWorker::FlushAttemptResult::Successful:
297 {
298 (void) worker.enqueue(parLogMessage);
299 break;
300 }
301
302 case ILogWorker::FlushAttemptResult::Unsuccessful:
303 {
304 if (options.flushPolicySettings.policies.test(FlushPolicy::Timed))
305 {
306 break;
307 }
308
309 JAUT_FALLTHROUGH;
310 }
311
312 case ILogWorker::FlushAttemptResult::Async:
313 {
314 const int timeout = options.bufferOverflowRetryTimeout;
315
316 if (timeout < 0)
317 {
318 // Do note that this is a dangerous combination for the reasons explained in the docs
319 // for jaut::Logger::Options::bufferOverflowRetryTimeout
320 jassert(!options.flushPolicySettings.policies.test(FlushPolicy::Levelled));
321
322 while (!worker.enqueue(parLogMessage))
323 {
324 if (flush_result != ILogWorker::FlushAttemptResult::Async)
325 {
326 (void) worker.tryFlush(parLogMessage);
327 }
328 }
329 }
330 else if (timeout > 0)
331 {
332 using Clock = std::chrono::steady_clock;
333 using TimePoint = std::chrono::time_point<Clock>;
334
335 using std::chrono::duration_cast;
336
337 const TimePoint start = Clock::now();
338
339 while (duration_cast<std::chrono::milliseconds>(Clock::now() - start).count() < timeout)
340 {
341 if (worker.enqueue(parLogMessage))
342 {
343 break;
344 }
345
346 if (flush_result != ILogWorker::FlushAttemptResult::Async)
347 {
348 (void) worker.tryFlush(parLogMessage);
349 }
350 }
351 }
352 else if (flush_result != ILogWorker::FlushAttemptResult::Unsuccessful)
353 {
354 (void) worker.enqueue(parLogMessage);
355 }
356
357 break;
358 }
359
360 // I have no clue how you got here, honestly
361 // It should be pretty much impossible to get here except you did something really weird,
362 // or you implemented your own worker returning this despite having been initialised at this point
363 case ILogWorker::FlushAttemptResult::NotReady: jassertfalse;
364 }
365 }
366 }
