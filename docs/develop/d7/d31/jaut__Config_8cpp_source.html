135 juce::String processConfigName(const juce::String &directory,
136 const jaut::IConfigParser *const parser,
137 const jaut::Config::Options &options)
138 {
139 if (!juce::File::isAbsolutePath(directory))
140 {
141 throw jaut::ConfigIOException("invalid file path'" + directory + "', directory must be an absolute path");
142 }
143
144 juce::File config_file = directory;
145
146 if (!config_file.exists() || !config_file.isDirectory())
147 {
148 throw jaut::ConfigIOException("invalid file path'" + directory
149 + "', file does not exist or is not a directory");
150 }
151
152 if (!options.fileName.isEmpty())
153 {
154 config_file = config_file.getChildFile(options.fileName);
155 }
156 else
157 {
158 const juce::String config_name = juce::String("config.") + ::deduceConfigName(parser).data();
159 config_file = config_file.getChildFile(config_name);
160 }
161
162 if (config_file.exists() && config_file.isDirectory())
163 {
164 throw jaut::ConfigIOException("invalid file path'" + config_file.getFullPathName()
165 + "', file exists but is a directory");
166 }
167
168 return config_file.getFullPathName();
169 }

298 bool Config::save() const
299 {
300 const juce::File &config_file = fullPath;
301
302 if (!config_file.getParentDirectory().exists())
303 {
304 if (!options.createFileIfNotExists)
305 {
306 return false;
307 }
308
309 if (const juce::Result result = config_file.create(); !result.wasOk())
310 {
311 throw ConfigIOException("config file could not be created, " + result.getErrorMessage()
312 + "(" + config_file.getFullPathName() + ")");
313 }
314 }
315
316 IConfigParser::ParseAttributes<const CategoryMap> attributes { config_file, properties, options };
317 parser->writeConfig(std::move(attributes));
318
319 return true;
320 }
