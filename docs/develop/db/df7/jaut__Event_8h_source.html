819 template<class ...HandlerArgs>
820 bool EventHandler<HandlerArgs...>::operator==(const EventHandler &parHandler) const noexcept
821 {
822 if (isAssociative() && parHandler.isAssociative())
823 {
824 return (std::get<juce::String>(data) == std::get<juce::String>(parHandler.data));
825 }
826
827 if (funcMode == parHandler.funcMode)
828 {
829 switch (funcMode)
830 {
831 case FuncMode::Free:
832 return (std::get<FreeFunction_t>(data) == std::get<FreeFunction_t>(parHandler.data));
833
834 case FuncMode::Member:
835 return (owner == parHandler.owner
836 && std::get<MemberDataArray>(data) == std::get<MemberDataArray>(parHandler.data));
837
838 case FuncMode::Chained:
839 return (owner == parHandler.owner);
840
841 case FuncMode::ClosureOrFunctor: break;
842 }
843 }
844
845 return false;
846 }
