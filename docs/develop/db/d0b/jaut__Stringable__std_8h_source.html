64 struct JAUT_API Stringable<std::array<T, N>>
65 {
66 JAUT_NODISCARD
67 static juce::String toString(const std::array<T, N> &object)
68 {
69 juce::StringArray elements;
70 elements.ensureStorageAllocated(static_cast<int>(object.size()));
71
72 for (const T &obj : object)

78 }
