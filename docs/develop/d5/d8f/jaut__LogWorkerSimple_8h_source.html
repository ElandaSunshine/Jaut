188 inline ILogWorker::FlushAttemptResult LogWorkerSimple<N, L>::tryFlush(const LogMessage &parLastMessage)
189 {
190 if (!logger)
191 {
192 return ILogWorker::FlushAttemptResult::NotReady;
193 }
194
195 const TimePoint now = Clock::now();
196 const int elapsed = static_cast<int>(std::chrono::duration_cast<std::chrono::seconds>(now - lastTime)
197 .count());
198
199 if ( flushBehaviour.policies.test(FlushPolicy::Instant)
200 || (flushBehaviour.policies.test(FlushPolicy::Timed) && elapsed >= flushBehaviour.interval)
201 || (flushBehaviour.policies.test(FlushPolicy::Filled) && isFull())
202 || (flushBehaviour.policies.test(FlushPolicy::Levelled) && parLastMessage.level >= flushBehaviour.level)
203 || (flushBehaviour.policies.test(FlushPolicy::Custom) && flushBehaviour.customPolicy(parLastMessage)))
204 {
205 flushInternal();
206 lastTime = now;
207
208 return ILogWorker::FlushAttemptResult::Successful;
209 }
210
211 return ILogWorker::FlushAttemptResult::Unsuccessful;
212 }
