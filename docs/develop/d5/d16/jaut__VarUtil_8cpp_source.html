39 VarUtil::VarTypeId VarUtil::getVarType(const juce::var &value) noexcept
40 {
41 if (value.isUndefined()) return VarTypeId::Undefined;
42 if (value.isString()) return VarTypeId::String;
43 if (value.isBool()) return VarTypeId::Bool;
44 if (value.isInt()) return VarTypeId::Int;
45 if (value.isInt64()) return VarTypeId::Int64;
46 if (value.isDouble()) return VarTypeId::Double;
47 if (value.isArray()) return VarTypeId::Array;
48 if (value.isObject()) return (value.getDynamicObject() ? VarTypeId::DynamicObject : VarTypeId::OtherObject);
49 if (value.isVoid()) return VarTypeId::Void;
50 if (value.isMethod()) return VarTypeId::Method;
51 if (value.isBinaryData()) return VarTypeId::BinaryData;
52
53 return VarTypeId::Unknown;
54 }
